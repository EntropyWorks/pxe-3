$def with (release,arch,ks) $ kernel = "images/pxeboot/vmlinuz" $ initrd = "images/pxeboot/initrd.img" $ proto = "http://"

LABEL CentOS $release $arch $ks
  KERNEL CentOS/$release/$arch/$kernel
  INITRD CentOS/$release/$arch/$initrd
  APPEND ramdisk_size=100000 ip=dhcp method=$proto$config.server/$config.ks_path/centos/$release/$arch/$ks
TEXT HELP
    Install CentOS $release $arch-$ks
ENDTEXT
