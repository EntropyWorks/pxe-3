$def with (os,release,arch,ks) $ kernel = "images/pxeboot/vmlinuz" $ initrd = "images/pxeboot/initrd.img" $ proto = "http://"

LABEL $os $release $arch $ks
  KERNEL $os/$release/$arch/$kernel
  KERNEL $os/$release/$arch/$initrd
  APPEND ramdisk_size=100000 ip=dhcp method=$proto$config.server/$config.ks_path/$os/$release/$arch/$ks
TEXT HELP
    Install $os $release $arch-$ks
ENDTEXT
